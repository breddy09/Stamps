    class WhatsNewModal < Browser::Modal
      def present?
        browser.img(css: 'img[src*=whatsnew]').present?
      end

      def close
        button = ElementWrapper.new browser.img(css: 'img[class*=x-tool-close]')
        5.times do
          button.click
          break unless button.present?
        end
      end

      def wait_until_present(*args)
        (ElementWrapper.new browser.img(css: 'img[src*=whatsnew]')).wait_until_present(*args)
      end

      def continue
        button = ElementWrapper.new(browser.span text: "Continue")
        10.times{
          button.click
          break unless button.present?
        }
      end
    end
